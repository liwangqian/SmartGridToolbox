33  /// @brief Convert spherical angles and magnitude to a vector.
34  ///
35  /// x -> north, y -> east, z -> directly up.
36  /// @param angs Struct containing spherical angles of the sun.
37  /// @param mag Magnitude of the vector.
38  /// @return The vector.
39  inline Array<double, 3> angsAndMagToVec(const SphericalAngles& angs, double mag)
40  {
41  using std::cos;
42  using std::sin;
43  return {{
44  mag * cos(angs.azimuth) * sin(angs.zenith), mag * sin(angs.azimuth) * sin(angs.zenith),
45  mag * cos(angs.zenith)
46  }};
47  }

49  /// @brief Convert spherical angles and projection to a vector.

51  /// @param proj projection of the vector onto a vertical vector.
52  /// @return The vector.
53  ///
54  /// x -> north, y -> east, z -> directly up.

57  inline Array<double, 3> angsAndProjToVec(const SphericalAngles& angs, double proj)
58  {
59  using std::cos;
60  using std::sin;

62  return angsAndMagToVec(angs, proj / cos(angs.zenith));
