120  void in(unsigned int n = 1)
121  {
122  const unsigned int delta = n * tabWidth_;
123  myIndentLevel_ += delta;
124  Log::indentLevel_ += delta;
125  }
126 
127  void out(unsigned int n = 1)
128  {
129  const unsigned int delta = std::min(n * tabWidth_, myIndentLevel_);
130  myIndentLevel_ -= delta;
131  Log::indentLevel_ -= delta;
132  }
