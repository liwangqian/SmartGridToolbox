187  /// @brief Soonest/smallest rank goes first. If equal rank and time, then sort on the id.
188  class ScheduledUpdatesCompare
189  {
190  public:
191  bool operator()(const std::pair<SimComponent*, Time>& lhs,
192  const std::pair<SimComponent*, Time>& rhs)
193  {
194  return ((lhs.second < rhs.second) ||
195  (lhs.second == rhs.second && lhs.first->rank() < rhs.first->rank()) ||
196  (lhs.second == rhs.second && lhs.first->rank() == rhs.first->rank() &&
197  (lhs.first->id() < rhs.first->id())));
198  }
199  };

201  // Smallest rank goes first. If equal rank, then sort on the id.
202  class ContingentUpdatesCompare
203  {
204  public:
205  bool operator()(const SimComponent* lhs, const SimComponent* rhs)
206  {
207  return ((lhs->rank() < rhs->rank()) ||
208  ((lhs->rank() == rhs->rank()) && (lhs->id() < rhs->id())));
209  }
